        function updateSimulasiHarga() {
            const selectedModel = document.getElementById('sim-motor-select').value;
            const motor = activeMotors.find(m => m.model === selectedModel);
            if (!motor) return;

            const price = Number(motor.harga);
            const minDp = price * 0.20; // minimal DP 20%
            const maxDp = price * 0.60; // maksimal DP 60%

            // Format OTR & Min DP text
            document.getElementById('sim-otr-text').innerText = formatRupiah(price);
            document.getElementById('sim-min-dp-text').innerText = formatRupiah(minDp);

            // Configure slider
            const slider = document.getElementById('sim-dp-slider');
            slider.min = minDp;
            slider.max = maxDp;
            slider.value = minDp; // Default at minimal DP

            document.getElementById('slider-min-label').innerText = "Min DP: " + formatRupiah(minDp);
            document.getElementById('slider-max-label').innerText = "Max DP: " + formatRupiah(maxDp);

            calculateAngsuran();
        }

        function calculateAngsuran() {
            const selectedModel = document.getElementById('sim-motor-select').value;
            const motor = activeMotors.find(m => m.model === selectedModel);
            if (!motor) return;

            const basePrice = Number(motor.harga);
            const dpVal = Number(document.getElementById('sim-dp-slider').value);

            // Tampilkan badge DP terpilih
            document.getElementById('sim-dp-badge').innerText = formatRupiah(dpVal);

            // Rencana Pokok Kredit
            const pokokKredit = basePrice - dpVal;
            
            // Estimasi Suku Bunga Flat Dealer (sekitar 1.5% flat per bulan)
            const bungaBulananRate = 0.015;
            const totalBunga = pokokKredit * bungaBulananRate * activeTenor;
            const totalPembalikan = pokokKredit + totalBunga;
            
            const cicilanBulanan = Math.ceil(totalPembalikan / activeTenor);

            document.getElementById('sim-angsuran-text').innerText = formatRupiah(cicilanBulanan) + " / bln";
        }

            const map = L.map('dealer-map').setView([latitude, longitude], 15);

                iconSize: [32, 32],
                iconAnchor: [16, 32],
                popupAnchor: [0, -32]

            const marker = L.marker([latitude, longitude], { icon: redMarkerIcon }).addTo(map);
